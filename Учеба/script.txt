#!/bin/bash

# Считываем имена пользователей с клавиатуры
echo "Введите имя первого пользователя:"
read user1

echo "Введите имя второго пользователя:"
read user2

# Получаем количество запущенных процессов для каждого пользователя
count_user1=$(ps -U $user1 | wc -l)
count_user2=$(ps -U $user2 | wc -l)

# Выводим результат сравнения
echo "Количество запущенных процессов у пользователя $user1: $count_user1"
echo "Количество запущенных процессов у пользователя $user2: $count_user2"

if [ $count_user1 -gt $count_user2 ]; then
    echo "У пользователя $user1 больше запущенных процессов."
elif [ $count_user2 -gt $count_user1 ]; then
    echo "У пользователя $user2 больше запущенных процессов."
else
    echo "У обоих пользователей одинаковое количество запущенных процессов."
fi
